\begin{itemize}
    \item \textbf{Division Algorithm: }$\forall a,b \in \mathbb{Z}, b \neq 0, \exists q,r \in \mathbb{Z} : a = bq + r, 0 \leq r < |b|$.
    \item \textbf{Inverse: }$\forall a \in \mathbb{Z}, \exists a^{-1} \in \mathbb{Z} : a \cdot a^{-1} \equiv 1 \pmod{m}.$. If $m$ is prime, then $a^{-1} \equiv a^{m-2} \pmod{m}$.
    \item \textbf{Euclide's Lemma: }$\forall a,b,p \in \mathbb{Z}, p \text{ is prime}, p|ab \Rightarrow p|a \text{ or } p|b$.
    \item \textbf{Bézout's identity: }$\forall x,y \in \mathbb{Z}^*, \exists a,b \in \mathbb{Z} : ax + by = $gcd$(a,b)$.
\end{itemize}
