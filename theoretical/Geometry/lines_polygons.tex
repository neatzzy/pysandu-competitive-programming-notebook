\subsection{Lines \& Polygons}

\begin{itemize}
    \item \textbf{Slope-Intercept: }$y = mx + c$
    \item \textbf{General Line Equation: }$Ax + By + C = 0$
    \item \textbf{Distance from point $(x_0, y_0)$ to line $Ax + By + C = 0$: }$d = \frac{|Ax_0 + By_0 + C|}{\sqrt{A^2 + B^2}}$
    \item \textbf{Shoelace Formula (Area of Polygon): }$A = \frac{1}{2} |\sum_{i=1}^{n-1}(x_i y_{i+1} - x_{i+1}y_i) + (x_n y_1 - x_1 y_n)|$ \\ OBS: All vertices must be in order, either clockwise or counterclockwise
    \item \textbf{Pick's Theorem (Area of Lattice Polygons): }$A = I + \frac{B}{2} - 1$, where $I$ is interior points and $B$ is boundary points
\end{itemize}
