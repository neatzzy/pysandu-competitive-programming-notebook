\subsection{Basic 2D Geometry}

\begin{itemize}
    \item \textbf{Distance Formula: }$\sqrt{\Delta x^2 + \Delta y^2}$
    \item \textbf{Vector Definition: }$\vec{v} = (x,y)$
    \item \textbf{Dot Product $(\vec{a} \cdot \vec{b})$: }$x_1 x_2 + y_1 y_2 = |\vec{a}| |\vec{b}| \cos \theta$ \\ Usage: Finding angles, checking perpendicularity $(\vec{a} \cdot \vec{b} = 0)$
    \item \textbf{Cross Product $(\vec{a} \times \vec{b}): $}$x_1 y_2 - y_1 x_2 = |\vec{a}| |\vec{b}| \sin \theta$ \\ Usage: 2D orientatiom, calculating signed area, collinearity $(\vec{a} \times \vec{b} = 0)$
    \item \textbf{Midpoint: }$M = (\frac{x_1 + x_2}{2}, \frac{y_1 + y_2}{2})$
\end{itemize}
