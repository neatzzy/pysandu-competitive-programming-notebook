\subsection{Graphs}

\begin{itemize}
    \item \textbf{Definition: }A graph $G = (V, E)$ has vertices $V$ and edges $E$. \textbf{Directed} (digraph) edges have direction; \textbf{undirected} edges do not.
    \item \textbf{Handshaking Lemma: }$\sum_{v \in V} \deg(v) = 2|E|$. In a digraph, $\sum \deg^+(v) = \sum \deg^-(v) = |E|$.
    \item \textbf{Tree: }A connected acyclic undirected graph. $n$ vertices $\Rightarrow$ exactly $n-1$ edges. Any two vertices are connected by a unique path.
    \item \textbf{Bipartite Graph: }Vertex set splits into $U$ and $V$ with all edges between the two sides. A graph is bipartite iff it contains no odd cycle. Check with BFS/DFS 2-coloring.
    \item \textbf{Euler Path/Circuit: }
          \begin{itemize}
              \item \textbf{Undirected:} Eulerian \textit{circuit} exists iff graph is connected and all vertices have even degree. Eulerian \textit{path} exists iff exactly 2 vertices have odd degree (start/end there).
              \item \textbf{Directed:} Eulerian \textit{circuit} exists iff graph is strongly connected and $\deg^+(v) = \deg^-(v)$ for all $v$. Eulerian \textit{path} exists iff one vertex has $\deg^+ - \deg^- = 1$ (start) and one has $\deg^- - \deg^+ = 1$ (end), all others balanced.
          \end{itemize}
    \item \textbf{Hamiltonian Path/Cycle: }Visits each vertex exactly once. NP-hard in general; solvable with bitmask DP in O($2^N \cdot N^2$) for small $N$.
    \item \textbf{Planar Graph: }Can be drawn without edge crossings. By Kuratowski's theorem, planar iff it contains no subdivision of $K_5$ or $K_{3,3}$. Euler's formula: $V - E + F = 2$ (connected planar graph).
    \item \textbf{Hall's Marriage Theorem: }A bipartite graph $(U, V)$ has a perfect matching iff for every $S \subseteq U$, $|N(S)| \geq |S|$.
    \item \textbf{Mantel's Theorem: }Triangle-free graph on $n$ vertices has at most $\lfloor n^2/4 \rfloor$ edges.
    \item \textbf{Four Color Theorem: }Any planar graph is 4-colorable. Bipartite graphs are 2-colorable. Cycle graphs: 2-colorable if even, 3-colorable if odd.
\end{itemize}
